ConversationBufferMemory: Stores the complete, raw history (Buffer).
ConversationSummaryMemory: Stores a continuously updated summary of the conversation (Summarization).
ConversationBufferWindowMemory: Stores only the last $K$ interactions (Windowing).


CREATE TABLE ollama_chat_history (
    -- Primary Key: Unique ID for each message/row
    id INT NOT NULL AUTO_INCREMENT,
    
    -- Session ID: The unique identifier that groups messages into a conversation
    -- LangChain will filter on this column to load the correct history.
    session_id VARCHAR(255) NOT NULL,
    
    -- Message Content: Stores the serialized chat message object (HumanMessage, AIMessage, etc.)
    -- TEXT or MEDIUMTEXT is needed to handle longer conversation messages.
    message TEXT NOT NULL,
    
    -- Constraints and Indices
    PRIMARY KEY (id),
    -- Indexing the session_id is CRITICAL for fast history retrieval
    INDEX idx_session_id (session_id) 
    
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
